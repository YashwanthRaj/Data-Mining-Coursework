Overview¶

We will be doing some clustering of zip codes based on their demographic features

The Census Bureau and other government agencies publish a lot of information about these, but it can take some effort to find and merge them all. There is a handy library called uszipcode that has compiled a number of these together. It has it's own API for interacting with the data, but it is backed by a sqlite database that it downloads. It will be simpler for us to just ingest the data from the sqlite database, rather than the API. In order to download the data, we need to first import it and specify that we want the comprehensive version of the data.

## This is a quickly put together set of cleaners for the columns that have a complex setup.

def extract(x):
    if x is None:
        return None
    else:
        extracted = json.loads(zlib.decompress(x).decode("utf-8"))
        return extracted


age_lookup = {'0': 'Under 5', '1': '5 to 9', '2': '10 to 14', '3': '15 to 19',
              '4': '20 to 24','5': '25 to 29','6': '30 to 34','7': '35 to 39','8': '40 to 44',
              '9': '45 to 49','10': '50 to 54','11': '55 to 59','12': '60 to 64','13': '65 to 69',
              '14': '70 to 74','15': '75 to 79','16': '80 to 84','17': 'above 85'}

age_lookup = {k:'population_by_age_'+'_'.join(v.split(' ')) for k,v in age_lookup.items()}

commute_mapping = {'< 10': 'commute_lt_10',
                   '10-19': 'commute_10_19',
                   '20-29': 'commute_20_29',
                   '30-39': 'commute_30_39',
                   '40-44': 'commute_40_44',
                   '45-59': 'commute_45_59',
                   '60-89': 'commute_60_89',
                   '90+': 'commute_ge_90'}

def clean_age(x):
    extracted = extract(x)
    if extracted is None:
        return None
    else:
        total = [i['values'] for i in extracted if i['key']=='Total'][0]
        
        data_dict = {age_lookup[str(i['x'])]:i['y'] for i in total}
        return data_dict

def clean_simple(x, prefix='population_by_race_', sep=' ', mapping=None):
    extracted = extract(x)
    if extracted is None:
        return None
    else:
        values = [i['values'] for i in extracted if i['key']=='Data'][0]
        if mapping is None:
            data_dict = {prefix+'_'.join(str(i['x']).split(sep)):i['y'] for i in values}
        else:
            data_dict = {mapping[str(i['x'])] :i['y'] for i in values}
        return data_dict

clean_race = lambda x: clean_simple(x, prefix='population_by_race_')
clean_education = lambda x: clean_simple(x, prefix='education_')
travel_time_to_work_in_minutes = lambda x: clean_simple(x, mapping=commute_mapping)

for col in encoded_fields:
    print(col)
    


df['population_by_age'] = df['population_by_age'].map(clean_age)
df['population_by_race'] = df['population_by_race'].map(clean_race)
df['educational_attainment_for_population_25_and_over'] = df['educational_attainment_for_population_25_and_over'].map(clean_education)
df['travel_time_to_work_in_minutes'] = df['travel_time_to_work_in_minutes'].map(travel_time_to_work_in_minutes)

ages = pd.json_normalize(df['population_by_age']) 
races = pd.json_normalize(df['population_by_race']) 
education = pd.json_normalize(df['educational_attainment_for_population_25_and_over'])
commute = pd.json_normalize(df['travel_time_to_work_in_minutes'])

df_merged = pd.concat([df, ages, races, education, commute], axis=1)
df = df_merged.drop(encoded_fields, axis=1)

def plot_silhouette_single(X, kmeans_model, plt):
    cluster_labels = kmeans_model.labels_
    n_clusters = len(np.unique(cluster_labels))

    silhouette_vals = silhouette_samples(X, cluster_labels)
    
    y_lower, y_upper = 0, 0
    yticks = []
    
    plt.figure(figsize=(8, 6)) 
    
    for i in range(n_clusters):
        ith_cluster_silhouette_vals = silhouette_vals[cluster_labels == i]
        ith_cluster_silhouette_vals.sort()
        
        y_upper += len(ith_cluster_silhouette_vals)
        plt.fill_betweenx(np.arange(y_lower, y_upper),
                          0, ith_cluster_silhouette_vals,
                          alpha=0.7)
        
        yticks.append((y_lower + y_upper) / 2)
        y_lower = y_upper

    plt.axvline(x=np.mean(silhouette_vals), color="red", linestyle="--")
    plt.yticks(yticks, range(n_clusters))
    plt.ylabel('Cluster')
    plt.xlabel('Silhouette Coefficient')
    plt.title(f'Silhouette Plot for {n_clusters} Clusters')
    plt.tight_layout()
    plt.show()

# unscaled data
plot_silhouette_single(X, kmeans, plt)

# scaled data
plot_silhouette_single(scaled_X2_train, kmeans2, plt)
